# Plan 06: Global Rules Sync

**Version**: v2.9.0 | **Date**: 2026-03-20 | **Status**: Ready for Implementation

---

## 🎯 Mục Tiêu

Thêm tính năng **Global Rules Sync** — cho phép quản lý và đồng bộ rules của AI coding tools thông qua sync-repo, tương tự như cách đang sync skills, workflows, và MCP servers.

**Ý tưởng cốt lõi:**
- User tạo rule files trong `.agents/rules/global/` trong sync-repo
- Khi chạy `agentools update`, package sẽ phát hiện và sync rules vào đúng thư mục của từng platform

**Tại sao cần feature này?**
- Hiện tại, rules của mỗi tool phải configure thủ công riêng lẻ trên từng máy
- Với global rules sync, user chỉ cần maintain 1 nơi (`sync-repo/.agents/rules/global/`) → auto-deploy đến tất cả platforms đã cài

Platform chia làm 2 nhóm:

### Nhóm 1: Multi-file folder (Claude Code, Cursor)
- **Platforms**: Claude Code (`~/.claude/rules/`), Cursor (`~/.cursor/rules/`)
- **Cách xử lý**: Copy **từng file** `.md` từ `.agents/rules/global/` vào rules folder (không merge)
- **Naming**: Giữ nguyên tên file gốc
- **Lý do**: Cả 2 platforms hỗ trợ multi-file rules folder với path-scoped rules (YAML frontmatter)
- **Ưu điểm**: Tổ chức tốt hơn, dễ maintain, mỗi topic 1 file

### Nhóm 2: Single-file (Antigravity, Windsurf, Codex CLI)
- **Platforms**: Antigravity (`~/.gemini/GEMINI.md`), Windsurf (`~/.codeium/windsurf/memories/global_rules.md`), Codex CLI (`~/.codex/AGENTS.md`)
- **Cách xử lý**: Merge tất cả `*.md` files trong `.agents/rules/global/` thành **1 file duy nhất**
- **Separator**: Dùng `\n\n---\n\n` giữa các file
- **Header comment**: Thêm `<!-- Generated by agentools - Do not edit manually -->` ở đầu file
- **Safety**: Không overwrite file hiện có trừ khi `--force` (trừ khi file do agentools tạo)

### Copilot — Skip
- Không có clear global rules path cross-platform
- **Quyết định**: Skip, ghi chú trong README

function ensureRulesDir(platform) {
  if (platform.rulesType !== "folder" || !platform.rulesPath) {
    return null;

### Phase 2: Tạo `rules-installer.js`

**File**: `package/scripts/rules-installer.js` ← **NEW FILE**

Module mới hoàn toàn, xử lý toàn bộ logic install rules.

```javascript
/**
 * Rules Installer Module
 * Syncs global rules from .agents/rules/global/ to AI platform directories
 */

const fs = require("fs");
const path = require("path");
const platforms = require("./platforms");

// Path tới rules trong cached sync-repo
const CACHE_DIR = path.join(platforms.HOME, ".agentools-cache");
const REPO_RULES_DIR = path.join(CACHE_DIR, ".agents", "rules", "global");

// Comment marker để identify agentools-managed content
const MANAGED_HEADER = "<!-- Generated by agentools - Do not edit manually -->\n";
const SECTION_SEPARATOR = "\n\n---\n\n";

/**
 * Get all .md files from global rules directory
 * @returns {string[]} Sorted array of absolute file paths
 */
function getGlobalRuleFiles() {
  if (!fs.existsSync(REPO_RULES_DIR)) {
    return [];
  }
  return fs
    .readdirSync(REPO_RULES_DIR)
    .filter((f) => f.endsWith(".md"))
    .sort()
    .map((f) => path.join(REPO_RULES_DIR, f));
}

/**
 * Merge multiple .md files into a single string
 * @param {string[]} files - Array of absolute file paths
 * @returns {string} Merged content
 */
function mergeRuleFiles(files) {
  if (files.length === 0) return "";
  const contents = files.map((f) => fs.readFileSync(f, "utf-8").trim());
  return MANAGED_HEADER + contents.join(SECTION_SEPARATOR) + "\n";
}

/**
 * Install rules for a platform that uses a single file (file-type)
 * @param {Object} platform
 * @param {string} mergedContent
 * @param {boolean} force
 */
function installRulesToFile(platform, mergedContent, force) {
  const rulesPath = platform.rulesPath;
  if (!rulesPath) return { installed: false, reason: "No rulesPath" };

  // Ensure parent directory exists
  const parentDir = path.dirname(rulesPath);
  if (!fs.existsSync(parentDir)) {
    fs.mkdirSync(parentDir, { recursive: true });
  }

  // Check if file already exists and is managed by agentools
  if (fs.existsSync(rulesPath) && !force) {
    const existing = fs.readFileSync(rulesPath, "utf-8");
    if (!existing.startsWith(MANAGED_HEADER)) {
      // File exists but NOT managed by us → skip to avoid overwriting user's rules
      return { installed: false, reason: "File exists and not managed by agentools (use --force)" };
    }
  }

  fs.writeFileSync(rulesPath, mergedContent, "utf-8");
  return { installed: true };
}

/**
 * Install rules for a platform that uses a folder (folder-type like Cursor)
 * @param {Object} platform
 * @param {string[]} ruleFiles
 * @param {boolean} force
 */
function installRulesToFolder(platform, ruleFiles, force) {
  const rulesDir = platform.rulesPath;
  if (!rulesDir) return { installed: false, count: 0 };

  if (!fs.existsSync(rulesDir)) {
    fs.mkdirSync(rulesDir, { recursive: true });
  }

  let installedCount = 0;
  let skippedCount = 0;

  for (const srcFile of ruleFiles) {
    const filename = path.basename(srcFile);
    const destPath = path.join(rulesDir, filename);

    if (fs.existsSync(destPath) && !force) {
      skippedCount++;
    } else {
      fs.copyFileSync(srcFile, destPath);
      installedCount++;
    }
  }

  return { installed: installedCount > 0, count: installedCount, skipped: skippedCount };
}

/**
 * Install global rules to all detected platforms
 * @param {Object} options - { force: boolean }
 */
function installRules(options = {}) {
  const { force = false } = options;

  const ruleFiles = getGlobalRuleFiles();
  if (ruleFiles.length === 0) {
    return { rulesCount: 0, platformsCount: 0, details: [] };
  }

  const mergedContent = mergeRuleFiles(ruleFiles);
  const detectedPlatforms = platforms.detectAll().map((p) => platforms.getByName(p.name));
  const SKIP_PLATFORMS = ["copilot"]; // Copilot không có clear global rules path

  const details = [];
  let totalPlatforms = 0;

  for (const platformObj of detectedPlatforms) {
    if (SKIP_PLATFORMS.includes(platformObj.name)) continue;
    if (!platformObj.rulesPath) continue;

    let result;
    if (platformObj.rulesType === "folder") {
      result = installRulesToFolder(platformObj, ruleFiles, force);
    } else {
      result = installRulesToFile(platformObj, mergedContent, force);
    }

    details.push({
      platform: platformObj.name,
      rulesPath: platformObj.rulesPath,
      ...result,
    });

    if (result.installed) totalPlatforms++;
  }

  return {
    rulesCount: ruleFiles.length,
    platformsCount: totalPlatforms,
    details,
  };
}

module.exports = {
  installRules,
  getGlobalRuleFiles,
  mergeRuleFiles,
  installRulesToFile,
  installRulesToFolder,
  REPO_RULES_DIR,
  MANAGED_HEADER,
};
```

### ✅ Low Risk
- Tính năng hoàn toàn opt-in: chỉ hoạt động khi `.agents/rules/global/` có files
- Backward compatible: không breaking changes

- **Per-platform rule selection**: Tất cả platforms nhận cùng rules → không filter
